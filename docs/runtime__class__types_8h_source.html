161struct OperandStack {
162 std::vector<Slot> stack;
163
164 // --- Push de 32 bits ---
165 void push_int(int32_t v) { stack.push_back(static_cast<u4>(v)); }
166
167 // --- Push de 64 bits (2 slots) ---
168 void push_long(int64_t v) {
169 u4 high = static_cast<u4>((v >> 32) & 0xFFFFFFFF);
170 u4 low = static_cast<u4>(v & 0xFFFFFFFF);
171 stack.push_back(high);
172 stack.push_back(low);
173 }
174
175 // --- Pop de 32 bits ---
176 int32_t pop_int() {
177 if (stack.empty())
178 throw std::runtime_error("Operand stack underflow");
179 int32_t v = static_cast<int32_t>(stack.back());
180 stack.pop_back();
181 return v;
182 }
183
184 // --- Pop de 64 bits ---
185 int64_t pop_long() {
186 if (stack.size() < 2)
187 throw std::runtime_error("Operand stack underflow");
188 u4 low = stack.back();
189 stack.pop_back();
190 u4 high = stack.back();
191 stack.pop_back();
192 return (static_cast<int64_t>(high) << 32) | low;
193 }
194
195 // --- Referências ---
196 void push_ref(RuntimeObject *ref) {
197 // armazenar ponteiro como índice ou endereço truncado (32-bit)
198 stack.push_back(static_cast<u4>(reinterpret_cast<uintptr_t>(ref)));
199 }
200
201 RuntimeObject *pop_ref() {
202 if (stack.empty())
203 throw std::runtime_error("Operand stack underflow");
204 u4 bits = stack.back();
205 stack.pop_back();
206 return reinterpret_cast<RuntimeObject *>(static_cast<uintptr_t>(bits));
207 }
208
209 // --- Float / Double helpers ---
210 void push_float(float v) {
211 union {
212 float f;
213 u4 u;
214 } conv;
215 conv.f = v;
216 stack.push_back(conv.u);
217 }
218
219 float pop_float() {
220 if (stack.empty())
221 throw std::runtime_error("Operand stack underflow");
222 union {
223 float f;
224 u4 u;
225 } conv;
226 conv.u = stack.back();
227 stack.pop_back();
228 return conv.f;
229 }
230
231 void push_double(double v) {
232 union {
233 double d;
234 u8 u;
235 } conv;
236 conv.d = v;
237 push_long(static_cast<int64_t>(conv.u));
238 }
239
240 double pop_double() {
241 union {
242 double d;
243 u8 u;
244 } conv;
245 conv.u = static_cast<u8>(pop_long());
246 return conv.d;
247 }
248
249 bool empty() const { return stack.empty(); }
250 size_t size() const { return stack.size(); }
251};
